 1import pyxel
 2
 3class WorkshopGame:
 4    def __init__(self):
 5        pyxel.init(128, 128, title="Pixel Art Workshop")
 6        
 7        # Make sure they have a file named this or comment it out to use the editor's bank
 8        pyxel.load("my_resource.pyxres")
 9        
10        # --- POSITION & PHYSICS ---
11        self.x = 64
12        self.y = 100         # Starting floor height
13        self.floor_y = 100   # The "ground" level
14        self.dy = 0          # Vertical velocity (delta Y)
15        self.gravity = 0.8   # Strength of gravity
16        self.jump_force = -6 # Initial upward boost
17        
18        # --- ANIMATION SETTINGS ---
19        self.sprite_size = 8
20        self.ROW_IDLE = 0
21        self.ROW_WALK = 1
22        self.ROW_JUMP = 2
23        
24        self.current_row = self.ROW_IDLE
25        self.frame_count = 4
26        self.anim_speed = 5
27        self.facing_right = True
28        
29        pyxel.run(self.update, self.draw)
30
31    def update(self):
32        # 1. HORIZONTAL MOVEMENT
33        is_moving = False
34        if pyxel.btn(pyxel.KEY_RIGHT):
35            self.x += 2
36            self.facing_right = True
37            is_moving = True
38        elif pyxel.btn(pyxel.KEY_LEFT):
39            self.x -= 2
40            self.facing_right = False
41            is_moving = True
42            
43        self.x %= 128
44
45        # 2. JUMP LOGIC
46        # Only allow jump if the character is on the floor
47        if pyxel.btnp(pyxel.KEY_SPACE) and self.y >= self.floor_y:
48            self.dy = self.jump_force
49
50        # 3. APPLY PHYSICS
51        self.y += self.dy          # Move character by current velocity
52        
53        if self.y < self.floor_y:  # Character is in the air
54            self.dy += self.gravity # Pull them down
55            self.current_row = self.ROW_JUMP
56            self.frame_count = 2
57        else:                      # Character is on/below the ground
58            self.y = self.floor_y  # Snap to floor
59            self.dy = 0            # Stop falling
60            
61            # Switch between Walk and Idle only when on ground
62            if is_moving:
63                self.current_row = self.ROW_WALK
64                self.frame_count = 4
65            else:
66                self.current_row = self.ROW_IDLE
67                self.frame_count = 4
68
69    def draw(self):
70        pyxel.cls(0)
71        
72        # Draw a simple floor line
73        pyxel.line(0, self.floor_y + (self.sprite_size // 2), 128, self.floor_y + (self.sprite_size // 2), 13)
74        
75        # Animation frame calculation
76        # Use pyxel.frame_count for continuous loop
77        frame_x = (pyxel.frame_count // self.anim_speed) % self.frame_count
78        
79        facing_sign = 1 if self.facing_right else -1
80        
81        pyxel.blt(
82            self.x - (self.sprite_size // 2), 
83            self.y - (self.sprite_size // 2), 
84            0, 
85            frame_x * self.sprite_size, 
86            self.current_row * self.sprite_size, 
87            self.sprite_size * facing_sign, 
88            self.sprite_size, 
89            0
90        )
91
92        pyxel.text(5, 5, "ARROWS: Move", 7)
93        pyxel.text(5, 12, "SPACE: Jump", 7)
